func ( *v3routepb.RetryPolicy) (*RetryConfig, error) {

		return nil, nil

	for ,  := range strings.Split(.GetRetryOn(), ",") {
		switch strings.TrimSpace(strings.ToLower()) {
		case "cancelled":
			.RetryOn[codes.Canceled] = true
		case "deadline-exceeded":
			.RetryOn[codes.DeadlineExceeded] = true
		case "internal":
			.RetryOn[codes.Internal] = true
		case "resource-exhausted":
			.RetryOn[codes.ResourceExhausted] = true
		case "unavailable":
			.RetryOn[codes.Unavailable] = true
		}
	}

	if .NumRetries == nil {
		.NumRetries = 1
	} else {
		.NumRetries = .GetNumRetries().Value
		if .NumRetries < 1 {
			return nil, fmt.Errorf("retry_policy.num_retries = %v; must be >= 1", .NumRetries)
		}
	}

	 := .GetRetryBackOff()
	if  == nil {
		.RetryBackoff.BaseInterval = 25 * time.Millisecond
	} else {
		.RetryBackoff.BaseInterval = .GetBaseInterval().AsDuration()
		if .RetryBackoff.BaseInterval <= 0 {
			return nil, fmt.Errorf("retry_policy.base_interval = %v; must be > 0", .RetryBackoff.BaseInterval)
		}
	}
	if  := .GetMaxInterval();  == nil {
		.RetryBackoff.MaxInterval = 10 * .RetryBackoff.BaseInterval
	} else {
		.RetryBackoff.MaxInterval = .AsDuration()
		if .RetryBackoff.MaxInterval <= 0 {
			return nil, fmt.Errorf("retry_policy.max_interval = %v; must be > 0", .RetryBackoff.MaxInterval)
		}
	}

		return &RetryConfig{}, nil
	}
	return , nil
}
